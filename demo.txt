copying elemnts of one array to another array:
#include <stdio.h>
void main()
{
    int arr1[]={1,2,3,4,5};
    int arr2[]={};
    int *arr1ptr,*arr2ptr,i;
    arr1ptr=arr1;
    arr2ptr=arr2;
    for(i=0;i<5;i++)
    {
        *arr1ptr=*arr2ptr;
        arr1ptr=arr1ptr+i;
        arr2ptr=arr2ptr+i;
        printf("%d ",arr2[i]);
    }
}
